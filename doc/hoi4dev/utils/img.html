 20def hex2rgb(hex):
 21    if hex.startswith('#'):
 22        hex = hex[1:]
 23    return tuple(int(hex[i:i+2], 16) for i in (0, 2, 4))

 25def hoi4dev_auto_image(
 26    path,
 27    searches = ["icon", "default"],
 28    resource_type = "idea",
 29    resource_default = True,
 30    scale = (-1, -1),
 31    cache_key = None,
 32    force = False
 33):
 34    cache_path = pjoin(path, ".cache", f"{resource_type}_{cache_key}.dds" if cache_key else f"{resource_type}.dds")
 35    if (not force) and ExistFile(cache_path):
 36        return ImageLoad(cache_path)
 37    if isinstance(scale, str):
 38        scale = get_mod_config('img_scales')[scale]
 39    w, h = scale
 40    icon = None
 41    for search in searches:
 42        if icon is None:
 43            icon = ImageFind(pjoin(path, search))
 44        else:
 45            break
 46    if (icon is None) and resource_default:
 47        if resource_default==True: resource_default = f"default_{resource_type}"
 48        else: resource_default = f"default_{resource_default}"
 49        icon = ImageFind(pjoin("hoi4dev_settings", "imgs", "defaults", resource_default), find_default=False)
 50        assert (icon is not None), f"The default {resource_type} icon is not found in {path}!"
 51    if icon is not None:
 52        icon = ImageZoom(icon, w=w, h=h)
 53        if force or (not ExistFile(cache_path)):
 54            CreateFolder(pjoin(path, ".cache"))
 55            ImageSave(icon, cache_path, format='dds')
 56    return icon

 95def ImageSave(img, path, format=None, flip_tga=True, compression='dxt3'):
 96    '''
 97    Save image to the given path with specified format.
 98    Args:
 99        path: str. If it contains suffix, the suffix will be replaced with `format`, otherwise the suffix will be appended.
100        format: str. Like 'dds', 'tga', 'png', etc. If not set, will use the suffix of `path`.
101        flip_tga: bool. Flip image if it is a `tga` file (specialized for HoI4 country flags).
102        compression: str. Compression method. Default is 'dxt3'. Only works for `dds` files.
103    Return:
104        None
105    
106    Optimizations for HOI4:
107    - support `dds` compression with `dxt3` by default.
108    - support `tga` flipping.
109    '''
110    cloned = img.clone()
111    cloned.alpha_channel = "set"
112    if STABLE_EXPORT:
113        time.sleep((cloned.size[0]*cloned.size[1]/2073600)*2.0+0.2) # 4.0s for each 4K image
114    if format == 'dds' or ((format is None) and path.endswith('.dds')):
115        if compression is not None:
116            cloned.compression = compression
117    elif format == 'tga' or ((format is None) and path.endswith('.tga')):
118        if flip_tga:
119            cloned.flip()
120    elif format == 'png' or ((format is None) and path.endswith('.png')):
121        pass
122    else:
123        pass
124    if format and (not path.endswith(f".{format}")):
125        path = AsFormat(path, format)
126    CreateFile(path)
127    cloned.save(filename=path)

129def ImageFind(path, priority=['png','dds','tga'], find_default=True):
130    '''
131    Find image in the given path with specified priority.
132    Args:
133        path: str. Path to the image file without suffix.
134        priority: list. Priority of suffixes.
135        find_default: bool. If True, will find file named `default` and search in it.
136    Return:
137        image.Image. A `wand` image object. None if not found.
138    
139    Notice that the function first search by the given path (assuming suffix contained).
140    Otherwise, it will search by the given path with each suffix in `priority` appended.
141    Finally, if `find_default` is True, it will search by the given path with `default` appended.
142    '''
143    if ExistFile(path):
144        return ImageLoad(path)
145    for p in priority:
146        if ExistFile('.'.join([path, p])):
147            return ImageLoad('.'.join([path, p]))
148    if find_default:
149        default_path = '/'.join(path.split('/')[:-1])+'/'+'default'
150        for p in priority:
151            if ExistFile('.'.join([default_path, p])):
152                return ImageLoad('.'.join([default_path, p]))
153    return None

155def ImageCopy(src_path, tgt_path, priority=['png','dds','tga'], find_default=True, format=None, flip_tga=False):
156    '''
157    Copy image from `src_path` to `tgt_path` with specified priority and save format.
158    Args:
159        src_path: str. Path to the source image file.
160        tgt_path: str. Path to the target image file.
161        priority: list. Priority of suffixes.
162        find_default: bool. If True, will find file named `default` and search in it.
163        format: str. Like 'dds', 'tga', 'png', etc.
164        flip_tga: bool. Flip image if it is a `tga` file (specialized for HoI4 country flags).
165    Return:
166        None
167    '''
168    img = ImageFind(src_path, priority=priority, find_default=find_default)
169    if img is not None:
170        ImageSave(img, tgt_path, format=format, flip_tga=flip_tga)
171    else:
172        raise FileNotFoundError(f"Image not found: \"{src_path}\"!")

202def ImageZoom(img, r=1, w=-1, h=-1, behavior='max'):
203    '''
204    Zoom image to the given size and crop/extend it accordingly to fit the desired size `(w,h)`. The extend image will be centered and filled with transparent color.
205    Args:
206        img: image.Image. A `wand` image object.
207        r: float. Ratio.
208        w: int. Width. If a float is given, it will be converted to int first.
209        h: int. Height. If a float is given, it will be converted to int first.
210        behavior: str. 'max' or 'min'.
211    Return:
212        image.Image. The zoomed image.
213    
214    Please pay attention to the behavior of this function:
215    0. The ratio of the image will always be kept.
216    1. If the ratio `r` is given, the image will alwyas be zoomed by the ratio and then cropped/extended. `w` and `h` should be either not set, or both set, otherwise leads to an error.
217    2. If neither `w` nor `h` is given, the image will not be modified.
218    3. Otherwise, if only one of `w` and `h` is given, the image will be zoomed to the given size.
219    4. If both `w` and `h` are given, if behavior is 'max', the image will be zoomed until both sides are larger than the given size, and then cropped to the given size; if behavior is 'min', the image will be zoomed unti only one side is larger than the given size, and then extended to the given size.
220    Examples:
221        (5, 3) zoom with `r=2` will result in (10, 6).
222        (5, 3) zoom with `r=2, w=30, h=30` will result in (10, 6) and then extended to (30, 30).
223        (5, 3) zoom with `w=30` will result in (30, 18).
224        (5, 3) zoom with `h=30` will result in (50, 30).
225        (5, 3) zoom with `w=30, h=30, behavior='max'` will result in (50, 30) and then cropped to (30, 30).
226        (5, 3) zoom with `w=30, h=30, behavior='min'` will result in (30, 18) and then extended to (30, 30).
227    '''
228    cloned = img.clone(); w = int(w); h = int(h)
229    if r != 1:
230        cloned.resize(int(cloned.width*r), int(cloned.height*r))
231        assert ((w!=-1 and h!=-1) or (w==-1 and h==-1)), "When the ratio `r` is given, `w` and `h` should be either not set, or both set!"
232        return cloned if (w==-1 and h==-1) else ImageExtend(cloned, w, h)
233    if w == -1 and h == -1:
234        return cloned
235    w_zoom_ratio = w / img.width; w_w = w; w_h = int(img.height*w_zoom_ratio)
236    h_zoom_ratio = h / img.height; h_w = int(img.width*h_zoom_ratio); h_h = h
237    min_w = min(w_w, h_w); min_h = min(w_h, h_h)
238    max_w = max(w_w, h_w); max_h = max(w_h, h_h)
239    if w==-1:
240        cloned.resize(h_w, h_h)
241    elif h==-1:
242        cloned.resize(w_w, w_h)
243    else:
244        if behavior == 'max':
245            cloned.resize(max_w, max_h)
246            cloned.crop(width=w, height=h, gravity='center')
247        elif behavior == 'min':
248            cloned.resize(min_w, min_h)
249            cloned = ImageExtend(cloned, w, h)
250    return cloned

252def ImageRotate(img, angle):
253    '''
254    Rotate image to the given angle. The rotated image will be centered and filled with transparent color.
255    Args:
256        img: image.Image. A `wand` image object.
257        angle: float. Angle in degree (clockwise).
258    Return:
259        image.Image. The rotated image.
260    
261    Notice that the image will always be zoomed larger or smaller to fit the rotated image.
262    '''
263    rot_w = int(img.width*abs(cos(radians(angle))) + img.height*abs(sin(radians(angle))))
264    rot_h = int(img.width*abs(sin(radians(angle))) + img.height*abs(cos(radians(angle))))
265    bg = CreateBlankImage(rot_w, rot_h)
266    bg.composite(img, int((rot_w-img.width)/2), int((rot_h-img.height)/2))
267    bg.rotate(angle, background=Color('transparent'))
268    return bg

270def ImageShift(img, dw=0, dh=0):
271    '''
272    Shift image by the given offset. The image will be extended to (w+dw,h+dh) and the shifted image will be filled with transparent color.
273    Args:
274        img: image.Image. A `wand` image object.
275        dw: int. Offset along width. Positive value means shift right.
276        dh: int. Offset along height. Positive value means shift down.
277    Return:
278        image.Image. The shifted image.
279    '''
280    ext = ImageExtend(img, w=img.width+abs(dw)*2, h=img.height+abs(dh)*2)
281    ext.crop(left=max(-dw,0), top=max(-dh,0), width=img.width+abs(dw), height=img.height+abs(dh))
282    return ext

284def ImageComposite(imgs):
285    '''
286    Composite images together. The images will be centered and filled with transparent color.
287    Args:
288        imgs: list[image.Image]. A list of `wand` image objects.
289    Return:
290        image.Image. The composited image.
291    '''
292    assert len(imgs) > 0, "No image to composite!"

294    bg = CreateBlankImage(w, h)
295    for img in imgs:
296        bg.composite(img, gravity='center')
297    return bg

299def ImageColorTransfer(img, src_color, tgt_color, intensity=0.3):
300    '''
301    Transfer the color of the image from `src_color` to `tgt_color`.
302    Args:
303        img: image.Image. A `wand` image object.
304        src_color: str. Source color in hex format.
305        tgt_color: str. Target color in hex format.
306        intensity: float. Intensity of the color transfer (0.0-1.0). 1.0 means painting, while 0.0 means no change.
307    Return:
308        image.Image. The color transferred image.
309    '''
310    cloned = img.clone()
311    src_color = Color(src_color)
312    inv_src_color = Color(rgb2hex((255-src_color.red_int8, 255-src_color.green_int8, 255-src_color.blue_int8)))
313    alpha = Color(rgb2hex((int(255*intensity), int(255*intensity), int(255*intensity))))
314    cloned.colorize(Color(tgt_color), alpha)
315    cloned.colorize(inv_src_color, alpha)
316    return cloned

332def ImageMask(img, mask, color='transparent'):
333    '''
334    Mask the image with the given mask.
335    Args:
336        img: image.Image. A `wand` image object.
337        mask: image.Image. A `wand` image object, supposed to be black and white. White part will be kept.
338        color: str. Color of the background after masking (transparent by default).
339    Return:
340        image.Image. The masked image.
341    '''
342    cloned = img.clone()
343    bg = CreateBlankImage(img.width, img.height, color)
344    alpha = mask.clone()
345    alpha.transform_colorspace('gray')
346    alpha.alpha_channel = 'copy'
347    cloned.composite_channel('alpha', alpha, 'copy_alpha')
348    bg.composite(cloned)
349    return bg

398def CreateIntelAgencyImage(img):
399    '''
400    Convert a image to an intelligence agency image.
401    Args:
402        img: image.Image. A `wand` image object.
403    Return:
404        image.Image. The country event image.
405    '''
406    W, H = get_mod_config('img_scales')['intel_agency_full']

416        bg.composite(shd, gravity='east')
417    bg.composite(img, gravity='east')
418    return bg

420def CreateBlueprintImage(img, color='white', radius=10.0, bg_color='blue', threshold=0.15, add_grid=6, grid_color='white', grid_opacity=0.2, grid_width=None):
421    '''
422    Convert a image to a blueprint image.
423    Args:
424        img: image.Image. A `wand` image object.
425        color: str. Color ('white' by default). Use by the blueprint lines.
426        radius: int. Width of the blueprint lines.
427        bg_color: str. Color (transparent by default).
428        threshold: float. Threshold for edge detection. The lower threshold will be `0.7*threshold` and the upper threshold will be `1.3*threshold`.
429        add_grid: int. Whether to add a grid to the blueprint. If an integer > 0, the grid will be added with the given size. If 0, no grid will be added.
430        grid_color: str. Color ('white' by default). Use by the grid.
431        grid_opacity: float. Opacity of the grid.
432        grid_width: int. Width of the grid. If None, will auto to 1/10 of the grid size.
433    Return:
434        image.Image. The blueprint image.
435    '''
436    lines = img.clone()
437
438    # Step 1: Extract edges
439    lines.transform_colorspace('gray')
440    lines.gaussian_blur(radius=radius/3)
441    lower = max(0, 0.7*threshold)
442    upper = min(1, 1.3*threshold)
443    lines.canny(radius=radius, lower_percent=lower, upper_percent=upper)
444    
445    # Now we have a image of white lines on a black background
446    # Step 2: Extract the white lines, and make black background transparent
447    lines.transparent_color(Color('white'), alpha=1)
448    lines.alpha_channel = 'remove'
449    lines.transparent_color(Color('black'), alpha=0)
450    lines.alpha_channel = 'set'
451    
452    # Step 3: Thickening the lines by dilation
453    lines.morphology('dilate', 'octagon', iterations=1)
454    
455    # Step 4: Change the color of the lines
456    if color != 'white':
457        lines.colorize(Color(color), Color('white'))
458    
459    # Step 4: Composite the lines over a blue background
460    bg = CreateBlankImage(img.width, img.height, bg_color)
461    if add_grid > 0:
462        grid = CreateBlankImage(img.width, img.height, 'transparent')
463        with Drawing() as draw:
464            draw.fill_color = Color(grid_color)
465            draw.fill_opacity = grid_opacity
466            draw.stroke_color = Color(grid_color)
467            draw.stroke_opacity = grid_opacity
468            if grid_width is None:
469                grid_width = add_grid // 10
470            draw.stroke_width = int(grid_width)
471            for x in range(add_grid//2, img.width, add_grid):
472                draw.line((x, 0), (x, img.height))
473            for y in range(add_grid//2, img.height, add_grid):
474                draw.line((0, y), (img.width, y))
475            draw(grid)
476        bg.composite(grid, 0, 0)
477    bg.composite(lines, 0, 0)
478    return bg

def ImageZoom(img, r=1, w=-1, h=-1, behavior='max'): 
203def ImageZoom(img, r=1, w=-1, h=-1, behavior='max'):
204    '''
205    Zoom image to the given size and crop/extend it accordingly to fit the desired size `(w,h)`. The extend image will be centered and filled with transparent color.
206    Args:
207        img: image.Image. A `wand` image object.
208        r: float. Ratio.
209        w: int. Width. If a float is given, it will be converted to int first.
210        h: int. Height. If a float is given, it will be converted to int first.
211        behavior: str. 'max' or 'min'.
212    Return:
213        image.Image. The zoomed image.
214    
215    Please pay attention to the behavior of this function:
216    0. The ratio of the image will always be kept.
217    1. If the ratio `r` is given, the image will alwyas be zoomed by the ratio and then cropped/extended. `w` and `h` should be either not set, or both set, otherwise leads to an error.
218    2. If neither `w` nor `h` is given, the image will not be modified.
219    3. Otherwise, if only one of `w` and `h` is given, the image will be zoomed to the given size.
220    4. If both `w` and `h` are given, if behavior is 'max', the image will be zoomed until both sides are larger than the given size, and then cropped to the given size; if behavior is 'min', the image will be zoomed unti only one side is larger than the given size, and then extended to the given size.
221    Examples:
222        (5, 3) zoom with `r=2` will result in (10, 6).
223        (5, 3) zoom with `r=2, w=30, h=30` will result in (10, 6) and then extended to (30, 30).
224        (5, 3) zoom with `w=30` will result in (30, 18).
225        (5, 3) zoom with `h=30` will result in (50, 30).
226        (5, 3) zoom with `w=30, h=30, behavior='max'` will result in (50, 30) and then cropped to (30, 30).
227        (5, 3) zoom with `w=30, h=30, behavior='min'` will result in (30, 18) and then extended to (30, 30).
228    '''
229    cloned = img.clone(); w = int(w); h = int(h)
230    if r != 1:
231        cloned.resize(int(cloned.width*r), int(cloned.height*r))
232        assert ((w!=-1 and h!=-1) or (w==-1 and h==-1)), "When the ratio `r` is given, `w` and `h` should be either not set, or both set!"
233        return cloned if (w==-1 and h==-1) else ImageExtend(cloned, w, h)
234    if w == -1 and h == -1:
235        return cloned
236    w_zoom_ratio = w / img.width; w_w = w; w_h = int(img.height*w_zoom_ratio)
237    h_zoom_ratio = h / img.height; h_w = int(img.width*h_zoom_ratio); h_h = h
238    min_w = min(w_w, h_w); min_h = min(w_h, h_h)
239    max_w = max(w_w, h_w); max_h = max(w_h, h_h)
240    if w==-1:
241        cloned.resize(h_w, h_h)
242    elif h==-1:
243        cloned.resize(w_w, w_h)
244    else:
245        if behavior == 'max':
246            cloned.resize(max_w, max_h)
247            cloned.crop(width=w, height=h, gravity='center')
248        elif behavior == 'min':
249            cloned.resize(min_w, min_h)
250            cloned = ImageExtend(cloned, w, h)
251    return cloned

Zoom image to the given size and crop/extend it accordingly to fit the desired size (w,h). The extend image will be centered and filled with transparent color.

def CreateBlueprintImage( img, color='white', radius=10.0, bg_color='blue', threshold=0.15, add_grid=6, grid_color='white', grid_opacity=0.2, grid_width=None): 
421def CreateBlueprintImage(img, color='white', radius=10.0, bg_color='blue', threshold=0.15, add_grid=6, grid_color='white', grid_opacity=0.2, grid_width=None):
422    '''
423    Convert a image to a blueprint image.
424    Args:
425        img: image.Image. A `wand` image object.
426        color: str. Color ('white' by default). Use by the blueprint lines.
427        radius: int. Width of the blueprint lines.
428        bg_color: str. Color (transparent by default).
429        threshold: float. Threshold for edge detection. The lower threshold will be `0.7*threshold` and the upper threshold will be `1.3*threshold`.
430        add_grid: int. Whether to add a grid to the blueprint. If an integer > 0, the grid will be added with the given size. If 0, no grid will be added.
431        grid_color: str. Color ('white' by default). Use by the grid.
432        grid_opacity: float. Opacity of the grid.
433        grid_width: int. Width of the grid. If None, will auto to 1/10 of the grid size.
434    Return:
435        image.Image. The blueprint image.
436    '''
437    lines = img.clone()
438
439    # Step 1: Extract edges
440    lines.transform_colorspace('gray')
441    lines.gaussian_blur(radius=radius/3)
442    lower = max(0, 0.7*threshold)
443    upper = min(1, 1.3*threshold)
444    lines.canny(radius=radius, lower_percent=lower, upper_percent=upper)
445    
446    # Now we have a image of white lines on a black background
447    # Step 2: Extract the white lines, and make black background transparent
448    lines.transparent_color(Color('white'), alpha=1)
449    lines.alpha_channel = 'remove'
450    lines.transparent_color(Color('black'), alpha=0)
451    lines.alpha_channel = 'set'
452    
453    # Step 3: Thickening the lines by dilation
454    lines.morphology('dilate', 'octagon', iterations=1)
455    
456    # Step 4: Change the color of the lines
457    if color != 'white':
458        lines.colorize(Color(color), Color('white'))
459    
460    # Step 4: Composite the lines over a blue background
461    bg = CreateBlankImage(img.width, img.height, bg_color)
462    if add_grid > 0:
463        grid = CreateBlankImage(img.width, img.height, 'transparent')
464        with Drawing() as draw:
465            draw.fill_color = Color(grid_color)
466            draw.fill_opacity = grid_opacity
467            draw.stroke_color = Color(grid_color)
468            draw.stroke_opacity = grid_opacity
469            if grid_width is None:
470                grid_width = add_grid // 10
471            draw.stroke_width = int(grid_width)
472            for x in range(add_grid//2, img.width, add_grid):
473                draw.line((x, 0), (x, img.height))
474            for y in range(add_grid//2, img.height, add_grid):
475                draw.line((0, y), (img.width, y))
476            draw(grid)
477        bg.composite(grid, 0, 0)
478    bg.composite(lines, 0, 0)
479    return bg

 38def ImageSave(img, path, format=None, flip_tga=True, compression='dxt3'):
 39    '''
 40    Save image to the given path with specified format.
 41    Args:
 42        path: str. If it contains suffix, the suffix will be replaced with `format`, otherwise the suffix will be appended.
 43        format: str. Like 'dds', 'tga', 'png', etc. If not set, will use the suffix of `path`.
 44        flip_tga: bool. Flip image if it is a `tga` file (specialized for HoI4 country flags).
 45        compression: str. Compression method. Default is 'dxt3'. Only works for `dds` files.
 46    Return:
 47        None
 48    
 49    Optimizations for HOI4:
 50    - support `dds` compression with `dxt3` by default.
 51    - support `tga` flipping.
 52    '''
 53    cloned = img.clone()
 54    if format == 'dds' or ((format is None) and path.endswith('.dds')):
 55        if compression is not None:
 56            cloned.compression = compression
 57    elif format == 'tga' or ((format is None) and path.endswith('.tga')):
 58        if flip_tga:
 59            cloned.flip()
 60    elif format == 'png' or ((format is None) and path.endswith('.png')):
 61        pass
 62    else:
 63        pass
 64    if format and (not path.endswith(f".{format}")):
 65        path = AsFormat(path, format)
 66    cloned.save(filename=path)

 68def ImageFind(path, priority=['png','dds','tga'], find_default=True):
 69    '''
 70    Find image in the given path with specified priority.
 71    Args:
 72        path: str. Path to the image file without suffix.
 73        priority: list. Priority of suffixes.
 74        find_default: bool. If True, will find file named `default` and search in it.
 75    Return:
 76        image.Image. A `wand` image object. None if not found.
 77    
 78    Notice that the function first search by the given path (assuming suffix contained).
 79    Otherwise, it will search by the given path with each suffix in `priority` appended.
 80    Finally, if `find_default` is True, it will search by the given path with `default` appended.
 81    '''
 82    if ExistFile(path):
 83        return ImageLoad(path)
 84    for p in priority:
 85        if ExistFile('.'.join([path, p])):
 86            return ImageLoad('.'.join([path, p]))
 87    if find_default:
 88        default_path = '/'.join(path.split('/')[:-1])+'/'+'default'
 89        for p in priority:
 90            if ExistFile('.'.join([default_path, p])):
 91                return ImageLoad('.'.join([default_path, p]))
 92    return None

 94def ImageCopy(src_path, tgt_path, priority=['png','dds','tga'], find_default=True, format=None, flip_tga=False):
 95    '''
 96    Copy image from `src_path` to `tgt_path` with specified priority and save format.
 97    Args:
 98        src_path: str. Path to the source image file.
 99        tgt_path: str. Path to the target image file.
100        priority: list. Priority of suffixes.
101        find_default: bool. If True, will find file named `default` and search in it.
102        format: str. Like 'dds', 'tga', 'png', etc.
103        flip_tga: bool. Flip image if it is a `tga` file (specialized for HoI4 country flags).
104    Return:
105        None
106    '''
107    img = ImageFind(src_path, priority=priority, find_default=find_default)
108    if img is not None:
109        ImageSave(img, tgt_path, format=format, flip_tga=flip_tga)
110    else:
111        raise FileNotFoundError(f"Image not found: \"{src_path}\"!")

141def ImageZoom(img, r=1, w=-1, h=-1, behavior='max'):
142    '''
143    Zoom image to the given size and crop/extend it accordingly to fit the desired size `(w,h)`. The extend image will be centered and filled with transparent color.
144    Args:
145        img: image.Image. A `wand` image object.
146        r: float. Ratio.
147        w: int. Width. If a float is given, it will be converted to int first.
148        h: int. Height. If a float is given, it will be converted to int first.
149        behavior: str. 'max' or 'min'.
150    Return:
151        image.Image. The zoomed image.
152    
153    Please pay attention to the behavior of this function:
154    0. The ratio of the image will always be kept.
155    1. If the ratio `r` is given, the image will alwyas be zoomed by the ratio and then cropped/extended. `w` and `h` should be either not set, or both set, otherwise leads to an error.
156    2. If neither `w` nor `h` is given, the image will not be modified.
157    3. Otherwise, if only one of `w` and `h` is given, the image will be zoomed to the given size.
158    4. If both `w` and `h` are given, if behavior is 'max', the image will be zoomed until both sides are larger than the given size, and then cropped to the given size; if behavior is 'min', the image will be zoomed unti only one side is larger than the given size, and then extended to the given size.
159    Examples:
160        (5, 3) zoom with `r=2` will result in (10, 6).
161        (5, 3) zoom with `r=2, w=30, h=30` will result in (10, 6) and then extended to (30, 30).
162        (5, 3) zoom with `w=30` will result in (30, 18).
163        (5, 3) zoom with `h=30` will result in (50, 30).
164        (5, 3) zoom with `w=30, h=30, behavior='max'` will result in (50, 30) and then cropped to (30, 30).
165        (5, 3) zoom with `w=30, h=30, behavior='min'` will result in (30, 18) and then extended to (30, 30).
166    '''
167    cloned = img.clone(); w = int(w); h = int(h)
168    if r != 1:
169        cloned.resize(int(cloned.width*r), int(cloned.height*r))
170        assert ((w!=-1 and h!=-1) or (w==-1 and h==-1)), "When the ratio `r` is given, `w` and `h` should be either not set, or both set!"
171        return cloned if (w==-1 and h==-1) else ImageExtend(cloned, w, h)
172    if w == -1 and h == -1:
173        return cloned
174    w_zoom_ratio = w / img.width; w_w = w; w_h = int(img.height*w_zoom_ratio)
175    h_zoom_ratio = h / img.height; h_w = int(img.width*h_zoom_ratio); h_h = h
176    min_w = min(w_w, h_w); min_h = min(w_h, h_h)
177    max_w = max(w_w, h_w); max_h = max(w_h, h_h)
178    if w==-1:
179        cloned.resize(h_w, h_h)
180    elif h==-1:
181        cloned.resize(w_w, w_h)
182    else:
183        if behavior == 'max':
184            cloned.resize(max_w, max_h)
185            cloned.crop(width=w, height=h, gravity='center')
186        elif behavior == 'min':
187            cloned.resize(min_w, min_h)
188            cloned = ImageExtend(cloned, w, h)
189    return cloned

191def ImageRotate(img, angle):
192    '''
193    Rotate image to the given angle. The rotated image will be centered and filled with transparent color.
194    Args:
195        img: image.Image. A `wand` image object.
196        angle: float. Angle in degree.
197    Return:
198        image.Image. The rotated image.
199    
200    Notice that the image will always be zoomed larger or smaller to fit the rotated image.
201    '''
202    rot_w = int(img.width*abs(cos(radians(angle))) + img.height*abs(sin(radians(angle))))
203    rot_h = int(img.width*abs(sin(radians(angle))) + img.height*abs(cos(radians(angle))))
204    bg = CreateBlankImage(rot_w, rot_h)
205    bg.composite(img, int((rot_w-img.width)/2), int((rot_h-img.height)/2))
206    bg.rotate(angle, background=image.Color('transparent'))
207    return bg

209def ImageShift(img, dw=0, dh=0):
210    '''
211    Shift image by the given offset. The image will be extended to (w+dw,h+dh) and the shifted image will be filled with transparent color.
212    Args:
213        img: image.Image. A `wand` image object.
214        dw: int. Offset along width. Positive value means shift right.
215        dh: int. Offset along height. Positive value means shift down.
216    Return:
217        image.Image. The shifted image.
218    '''
219    ext = ImageExtend(img, w=img.width+abs(dw)*2, h=img.height+abs(dh)*2)
220    ext.crop(left=max(-dw,0), top=max(-dh,0), width=img.width+abs(dw), height=img.height+abs(dh))
221    return ext

270def CreateIntelAgencyImage(img):
271    '''
272    Convert a image to an intelligence agency image.
273    Args:
274        img: image.Image. A `wand` image object.
275    Return:
276        image.Image. The country event image.
277    '''
278    W, H = get_mod_config('img_scales')['intel_agency_full']
279    w, h = get_mod_config('img_scales')['intel_agency']
280    bg = CreateBlankImage(W, H)
281    img = ImageZoom(img, w=w, h=h)
282    bg.composite(img, gravity='west')
283    shd = img.clone()
284    shd.shadow(1.0, 1.0, 0, 0)
285    shd = ImageExtend(ImageZoom(shd, r=1.1), w, h)
286    # Thickening the shadow
287    for _ in range(8):
288        bg.composite(shd, gravity='east')
289    bg.composite(img, gravity='east')
290    return bg

143    '''
144    Zoom image to the given size and crop/extend it accordingly to fit the desired size `(w,h)`. The extend image will be centered and filled with transparent color.
145    Args:
146        img: image.Image. A `wand` image object.
147        r: float. Ratio.
148        w: int. Width. If a float is given, it will be converted to int first.
149        h: int. Height. If a float is given, it will be converted to int first.
150        behavior: str. 'max' or 'min'.
151    Return:
152        image.Image. The zoomed image.
153    
154    Please pay attention to the behavior of this function:
155    0. The ratio of the image will always be kept.
156    1. If the ratio `r` is given, the image will alwyas be zoomed by the ratio and then cropped/extended. `w` and `h` should be either not set, or both set, otherwise leads to an error.
157    2. If neither `w` nor `h` is given, the image will not be modified.
158    3. Otherwise, if only one of `w` and `h` is given, the image will be zoomed to the given size.
159    4. If both `w` and `h` are given, if behavior is 'max', the image will be zoomed until both sides are larger than the given size, and then cropped to the given size; if behavior is 'min', the image will be zoomed unti only one side is larger than the given size, and then extended to the given size.
160    Examples:
161        (5, 3) zoom with `r=2` will result in (10, 6).
162        (5, 3) zoom with `r=2, w=30, h=30` will result in (10, 6) and then extended to (30, 30).
163        (5, 3) zoom with `w=30` will result in (30, 18).
164        (5, 3) zoom with `h=30` will result in (50, 30).
165        (5, 3) zoom with `w=30, h=30, behavior='max'` will result in (50, 30) and then cropped to (30, 30).
166        (5, 3) zoom with `w=30, h=30, behavior='min'` will result in (30, 18) and then extended to (30, 30).
167    '''
168    cloned = img.clone(); w = int(w); h = int(h)
169    if r != 1:
170        cloned.resize(int(cloned.width*r), int(cloned.height*r))
171        assert ((w!=-1 and h!=-1) or (w==-1 and h==-1)), "When the ratio `r` is given, `w` and `h` should be either not set, or both set!"
172        return cloned if (w==-1 and h==-1) else ImageExtend(cloned, w, h)
173    if w == -1 and h == -1:
174        return cloned
175    w_zoom_ratio = w / img.width; w_w = w; w_h = int(img.height*w_zoom_ratio)
176    h_zoom_ratio = h / img.height; h_w = int(img.width*h_zoom_ratio); h_h = h
177    min_w = min(w_w, h_w); min_h = min(w_h, h_h)
178    max_w = max(w_w, h_w); max_h = max(w_h, h_h)
179    if w==-1:
180        cloned.resize(h_w, h_h)
181    elif h==-1:
182        cloned.resize(w_w, w_h)
183    else:
184        if behavior == 'max':
185            cloned.resize(max_w, max_h)
186            cloned.crop(width=w, height=h, gravity='center')
187        elif behavior == 'min':
188            cloned.resize(min_w, min_h)
189            cloned = ImageExtend(cloned, w, h)
190    return cloned

Zoom image to the given size and crop/extend it accordingly to fit the desired size (w,h). The extend image will be centered and filled with transparent color.

Arguments:
  • img: image.Image. A wand image object.
  • r: float. Ratio.
  • w: int. Width. If a float is given, it will be converted to int first.
  • h: int. Height. If a float is given, it will be converted to int first.
  • behavior: str. 'max' or 'min'.
Return:

image.Image. The zoomed image.

Please pay attention to the behavior of this function:

  1. The ratio of the image will always be kept.
  2. If the ratio r is given, the image will alwyas be zoomed by the ratio and then cropped/extended. w and h should be either not set, or both set, otherwise leads to an error.
  3. If neither w nor h is given, the image will not be modified.
  4. Otherwise, if only one of w and h is given, the image will be zoomed to the given size.
  5. If both w and h are given, if behavior is 'max', the image will be zoomed until both sides are larger than the given size, and then cropped to the given size; if behavior is 'min', the image will be zoomed unti only one side is larger than the given size, and then extended to the given size.
Examples:

(5, 3) zoom with h=30 will result in (50, 30). (5, 3) zoom with w=30, h=30, behavior='max' will result in (50, 30) and then cropped to (30, 30). 211    '''
212    Shift image by the given offset. The image will be extended to (w+dw,h+dh) and the shifted image will be filled with transparent color.
213    Args:

218        image.Image. The shifted image.
219    '''
220    ext = ImageExtend(img, w=img.width+abs(dw)*2, h=img.height+abs(dh)*2)

222    return ext
